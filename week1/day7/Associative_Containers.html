üéØ Introduction to Associative Containers

Why "Associative"?

These containers are called Associative Containers because they store data based on keys and keep elements sorted automatically.

Internal Implementation

  • Internally implemented using Self-Balancing Binary Search Tree (Red-Black Tree)
  • Automatically maintains sorted order
  • All major operations: O(log n)
  • Height-balanced tree ensures performance

Time Complexity

  • Insert: O(log n)
  • Search: O(log n)
  • Delete: O(log n)
  • Access: O(log n)

Required Header

#include <map>      // for map and multimap
#include <set>      // for set and multiset

üî∑ 1Ô∏è‚É£ MAP (std::map)

üìå What is Map?

  • Stores key-value pairs
  • Keys are unique (no duplicates)
  • Elements are sorted by key
  • Implemented using Red-Black Tree
  • Allows fast lookup, insertion, and deletion

üîπ Syntax

#include <map>
std::map<key_type, value_type> map_name;

// Example:
map<int, string> m;

üîπ All Important Member Functions

Complete Practical Example:

#include <iostream>
#include <map>
using namespace std;

int main() {
    // ===============================
    // 1Ô∏è‚É£ Declaration
    // ===============================
    map<int, string> m;  // Creates an empty map
    
    // ===============================
    // 2Ô∏è‚É£ insert() - Add elements
    // ===============================
    m.insert({1, "Apple"});     // Method 1: using pair
    m.insert({2, "Banana"});
    
    // Insert with return value check
    auto result = m.insert({3, "Cherry"});
    if (result.second)  // .second is bool (success?)
        cout << "Inserted successfully" << endl;
    
    // ===============================
    // 3Ô∏è‚É£ operator[] - Access/Insert
    // ===============================
    m[4] = "Mango";          // If key doesn't exist ‚Üí create it
    m[1] = "Avocado";        // If key exists ‚Üí update value
    
    // ===============================
    // 4Ô∏è‚É£ at() - Safe Access
    // ===============================
    cout << m.at(1) << endl;  // Returns value of key 1
    // m.at(100);  // Throws std::out_of_range exception
    
    // ===============================
    // 5Ô∏è‚É£ find() - Search for key
    // ===============================
    auto it = m.find(2);
    if (it != m.end()) {
        cout << "Found: " << it->first << " -> " << it->second << endl;
    }
    
    // ===============================
    // 6Ô∏è‚É£ size() - Number of elements
    // ===============================
    cout << "Size: " << m.size() << endl;
    
    // ===============================
    // 7Ô∏è‚É£ empty() - Check if empty
    // ===============================
    if (m.empty())
        cout << "Map is empty";
    else
        cout << "Map is not empty" << endl;
    
    // ===============================
    // 8Ô∏è‚É£ erase() - Remove elements
    // ===============================
    m.erase(2);               // Erase by key
    m.erase(m.find(3));      // Erase using iterator
    
    // ===============================
    // 9Ô∏è‚É£ count() - Check existence
    // ===============================
    if (m.count(1))           // Returns 1 if exists, 0 if not
        cout << "Key 1 exists" << endl;
    
    // ===============================
    // üîü lower_bound() & upper_bound()
    // ===============================
    m[5] = "Orange"; m[7] = "Grape"; m[10] = "Kiwi";
    
    auto lb = m.lower_bound(6);  // First element >= 6 (returns key 7)
    auto ub = m.upper_bound(7);  // First element > 7 (returns key 10)
    
    if (lb != m.end())
        cout << "lower_bound(6): " << lb->first << endl;
    
    // ===============================
    // 1Ô∏è‚É£1Ô∏è‚É£ clear() - Remove all
    // ===============================
    // m.clear();  // Removes all elements
    
    // ===============================
    // 1Ô∏è‚É£2Ô∏è‚É£ Traversal - Different ways
    // ===============================
    
    // Method 1: Range-based for loop (Recommended)
    for (auto p : m)
        cout << p.first << " -> " << p.second << endl;
    
    // Method 2: Using iterators
    for (auto it = m.begin(); it != m.end(); ++it)
        cout << it->first << " -> " << it->second << endl;
    
    // ===============================
    // 1Ô∏è‚É£3Ô∏è‚É£ Reverse Traversal
    // ===============================
    for (auto it = m.rbegin(); it != m.rend(); ++it)
        cout << it->first << " -> " << it->second << endl;
    
    return 0;
}

‚ö†Ô∏è Important Notes on map:

  • operator[] creates a new element if key doesn't exist (can be dangerous!)
  • at() throws exception if key not found (safer for read-only access)
  • Use find() or count() to check existence before accessing
  • Cannot modify keys directly; must erase and re-insert

üî∑ 2Ô∏è‚É£ SET (std::set)

üìå What is Set?

  • Stores only keys (no values)
  • Keys must be unique
  • Automatically sorted
  • No indexing (cannot access by set[0])
  • Fast lookup, insertion, and deletion

üîπ Syntax

#include <set>
std::set<data_type> set_name;

// Example:
set<int> s;

üîπ All Important Member Functions

#include <iostream>
#include <set>
using namespace std;

int main() {
    set<int> s;
    
    // insert() - Add elements
    s.insert(10);
    s.insert(20);
    s.insert(30);
    s.insert(10);  // Duplicate - ignored
    
    // size()
    cout << "Size: " << s.size() << endl;  // Output: 3
    
    // find() - Returns iterator
    auto it = s.find(20);
    if (it != s.end())
        cout << "Found: " << *it << endl;
    
    // count() - Returns 0 or 1
    cout << s.count(10) << endl;  // Output: 1 (exists)
    cout << s.count(100) << endl; // Output: 0 (doesn't exist)
    
    // lower_bound() & upper_bound()
    auto lb = s.lower_bound(15);  // First element >= 15 (returns 20)
    auto ub = s.upper_bound(20);  // First element > 20 (returns 30)
    
    // erase() - Remove element
    s.erase(10);              // Erase by value
    s.erase(s.find(20));      // Erase using iterator
    
    // Traversal
    for (auto x : s)
        cout << x << " ";  // Sorted order
    
    // Reverse traversal
    for (auto it = s.rbegin(); it != s.rend(); ++it)
        cout << *it << " ";
    
    // clear() - Remove all elements
    s.clear();
    
    return 0;
}

üí° Set Use Cases:

  • Remove duplicates from a collection
  • Maintain sorted unique elements
  • Fast membership testing
  • Finding ranges in sorted data

üî∑ 3Ô∏è‚É£ MULTIMAP (std::multimap)

üìå What is Multimap?

  • Same as map
  • Allows duplicate keys
  • No [] operator (because which value to return?)
  • Keys are sorted
  • Multiple values can exist for same key

üîπ Syntax

multimap<int, string> mm;

üîπ Important Member Functions

#include <iostream>
#include <map>
using namespace std;

int main() {
    multimap<int, string> mm;
    
    // insert() - Allows duplicates
    mm.insert({1, "Apple"});
    mm.insert({1, "Avocado"});   // Duplicate key allowed!
    mm.insert({2, "Banana"});
    mm.insert({1, "Apricot"});  // Another duplicate
    
    // size()
    cout << "Size: " << mm.size() << endl;  // Output: 4
    
    // count() - Returns number of occurrences
    cout << "Count of key 1: " << mm.count(1) << endl;  // Output: 3
    
    // find() - Returns first occurrence
    auto it = mm.find(1);
    if (it != mm.end())
        cout << "First occurrence: " << it->second << endl;
    
    // equal_range() - Get all values for a key
    auto range = mm.equal_range(1);
    // range.first ‚Üí iterator to first occurrence
    // range.second ‚Üí iterator to one past last occurrence
    
    cout << "All values for key 1: ";
    for (auto i = range.first; i != range.second; ++i)
        cout << i->second << " ";
    cout << endl;
    
    // erase() - Removes ALL occurrences of key
    mm.erase(1);  // Removes all 3 entries with key 1
    
    // To erase only one occurrence:
    mm.erase(mm.find(2));  // Erase using iterator
    
    // Traversal
    for (auto p : mm)
        cout << p.first << " -> " << p.second << endl;
    
    return 0;
}

‚ö†Ô∏è Key Differences from map:

  • No operator[] available
  • count() can return values > 1
  • erase(key) removes ALL occurrences
  • Use equal_range() to get all values for a key

üî∑ 4Ô∏è‚É£ MULTISET (std::multiset)

üìå What is Multiset?

  • Same as set
  • Allows duplicates
  • Sorted automatically
  • Cannot modify element after insertion

üîπ Syntax

multiset<int> ms;

üîπ Important Member Functions

#include <iostream>
#include <set>
using namespace std;

int main() {
    multiset<int> ms;
    
    // insert() - Allows duplicates
    ms.insert(5);
    ms.insert(5);  // Duplicate allowed
    ms.insert(10);
    ms.insert(5);  // Another duplicate
    ms.insert(20);
    
    // count() - Returns number of occurrences
    cout << "Count of 5: " << ms.count(5) << endl;  // Output: 3
    
    // find() - Returns iterator to first occurrence
    auto it = ms.find(5);
    if (it != ms.end())
        cout << "Found: " << *it << endl;
    
    // equal_range() - Get range of equal elements
    auto range = ms.equal_range(5);
    cout << "Elements equal to 5: ";
    for (auto i = range.first; i != range.second; ++i)
        cout << *i << " ";
    cout << endl;
    
    // erase() - Removes ALL occurrences
    ms.erase(5);  // Removes all three 5's
    
    // To erase only one occurrence:
    ms.insert(5); ms.insert(5);
    ms.erase(ms.find(5));  // Erase only one 5
    
    // Traversal (sorted order with duplicates)
    for (auto x : ms)
        cout << x << " ";
    
    return 0;
}

üí° Multiset Use Cases:

  • Frequency counting with sorting
  • Priority queue alternative (when duplicates needed)
  • Maintaining sorted data with duplicates
  • Sliding window problems

üî∑ 5Ô∏è‚É£ Iterators in Associative Containers

üìå Types of Iterators

Forward Iterators

begin()  // First element
end()    // Past-the-end

Reverse Iterators

rbegin() // Last element
rend()   // Before first

Const Iterators

cbegin()  // Const begin
cend()    // Const end
crbegin() // Const reverse
crend()   // Const reverse end

Complete Iterator Example

#include <iostream>
#include <map>
using namespace std;

int main() {
    map<int, string> m = {{1, "A"}, {2, "B"}, {3, "C"}};
    
    // Forward iteration
    cout << "Forward: ";
    for (auto it = m.begin(); it != m.end(); ++it)
        cout << it->first << " ";
    cout << endl;
    
    // Reverse iteration
    cout << "Reverse: ";
    for (auto it = m.rbegin(); it != m.rend(); ++it)
        cout << it->first << " ";
    cout << endl;
    
    // Range-based (recommended)
    cout << "Range-based: ";
    for (const auto& [key, value] : m)  // C++17 structured binding
        cout << key << "->" << value << " ";
    
    return 0;
}

‚ö†Ô∏è Iterator Invalidation

  • Insertion: Does NOT invalidate any iterators
  • Deletion: Only invalidates iterators to the deleted element
  • Unlike vectors, insertion doesn't invalidate iterators in associative containers

üî∑ 6Ô∏è‚É£ Advanced Concepts

1Ô∏è‚É£ Custom Comparator

By default, elements are sorted in ascending order. To change this:

#include <iostream>
#include <map>
#include <set>
#include <functional>
using namespace std;

int main() {
    // Descending order map
    map<int, string, greater<int>> m;
    m[3] = "C"; m[1] = "A"; m[2] = "B";
    
    for (auto p : m)
        cout << p.first << " ";  // Output: 3 2 1
    cout << endl;
    
    // Custom comparator for set
    auto cmp = [](int a, int b) { return a > b; };
    set<int, decltype(cmp)> s(cmp);
    s.insert(5); s.insert(2); s.insert(8);
    
    for (int x : s)
        cout << x << " ";  // Output: 8 5 2
    
    return 0;
}

2Ô∏è‚É£ Custom Objects as Keys

#include <iostream>
#include <map>
using namespace std;

struct Person {
    string name;
    int age;
    
    // Must define operator< for sorting
    bool operator<(const Person& other) const {
        return age < other.age;  // Sort by age
    }
};

int main() {
    map<Person, string> m;
    
    m[{"Alice", 25}] = "Engineer";
    m[{"Bob", 30}] = "Doctor";
    m[{"Charlie", 20}] = "Student";
    
    for (const auto& [person, job] : m)
        cout << person.name << " (" << person.age << "): " << job << endl;
    
    return 0;
}

4Ô∏è‚É£ emplace() vs insert()

üí° Performance Tip:

emplace() constructs the object in-place, avoiding extra copy/move operations.

map<int, string> m;

// insert() - creates temporary object first
m.insert({1, "Apple"});

// emplace() - constructs directly (more efficient)
m.emplace(2, "Banana");  // Slightly faster

map<int, string> m1 = {{1, "A"}};

üî∑ 8Ô∏è‚É£ When to Use What?

Use map when:

  • Each key has exactly one value
  • Need key-value mapping
  • Dictionary-like structure
  • Example: Student ID ‚Üí Name

Use multimap when:

  • One key can have multiple values
  • Need to group by key
  • Example: Date ‚Üí Events
  • Example: Category ‚Üí Products

Use set when:

  • Need unique sorted values
  • Remove duplicates
  • Fast membership testing
  • Example: Unique usernames

Use multiset when:

  • Duplicates allowed but sorted
  • Frequency counting
  • Priority queue with duplicates
  • Example: Leaderboard scores

Real-World Examples

Example 1: Frequency Counter

map<char, int> freq;
string s = "hello";
for (char c : s)
    freq[c]++;  // Counts: h=1, e=1, l=2, o=1

Example 2: Group Anagrams

map<string, vector<string>> anagrams;
vector<string> words = {"eat", "tea", "tan", "ate"};

for (string word : words) {
    string sorted = word;
    sort(sorted.begin(), sorted.end());
    anagrams[sorted].push_back(word);
}

Example 3: Range Queries

set<int> s = {1, 3, 5, 7, 9};

// Find all elements in range [4, 8]
auto start = s.lower_bound(4);  // Points to 5
auto end = s.upper_bound(8);    // Points to 9

for (auto it = start; it != end; ++it)
    cout << *it << " ";  // Output: 5 7

üî∑ 9Ô∏è‚É£ Common Mistakes to Avoid

‚ö†Ô∏è Mistake 1: Using [] on non-existent key

map<int, int> m;
cout << m[5];  // Creates entry with value 0!

// Solution: Use find() or at()
if (m.find(5) != m.end())
    cout << m[5];

‚ö†Ô∏è Mistake 2: Modifying elements while iterating

// WRONG: Iterator invalidation
for (auto it = m.begin(); it != m.end(); ++it) {
    m.erase(it->first);  // BUG!
}

// CORRECT: Use erase's return value
for (auto it = m.begin(); it != m.end(); ) {
    it = m.erase(it);  // Returns next valid iterator
}

‚ö†Ô∏è Mistake 3: Comparing map with ==

// This compares both keys AND values
map<int, int> m1 = {{1, 10}};
map<int, int> m2 = {{1, 20}};
cout << (m1 == m2);  // false (different values)

‚ö†Ô∏è Mistake 4: Forgetting about custom comparators

struct Point {
    int x, y;
    // MUST define operator< or provide custom comparator!
};

// This won't compile without operator<
set<Point> s;  // ERROR!

‚ö†Ô∏è Mistake 5: Using erase() with multimap/multiset

multiset<int> ms = {5, 5, 5, 10};
ms.erase(5);  // Removes ALL three 5's!

// To remove only one:
ms.erase(ms.find(5));  // Removes only one 5

üéØ Key Takeaways

  • All operations: O(log n) time complexity
  • Automatically sorted by keys (Red-Black Tree)
  • Use map for unique key-value pairs
  • Use set for unique sorted values
  • Add "multi" prefix to allow duplicates
  • No indexing - use iterators for access
  • Iterator invalidation only on deletion of that specific element
  • lower_bound() and upper_bound() for range queries
